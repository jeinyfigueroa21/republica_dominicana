305:       while($num < $total)

309:             'num' => $i + 1,
310:             'actual' => ($num == $this->offset)
311:          );
312: 
313:          if($num == $this->offset)
314:          {
315:             $actual = $i;
316:          }
317: 
318:          $i++;
319:          $num += FS_ITEM_LIMIT;
320:       }
321: 
322:       /// ahora descartamos
323:       foreach($paginas as $j => $value)
324:       {
325:          $enmedio = intval($i/2);
326: 
327:          /**
328:           * descartamos todo excepto la primera, la última, la de enmedio,
329:           * la actual, las 5 anteriores y las 5 siguientes
330:           */
331:          if( ($j>1 AND $j<$actual-5 AND $j!=$enmedio) OR ($j>$actual+5 AND $j<$i-1 AND $j!=$enmedio) )
332:          {
333:             unset($paginas[$j]);
334:          }
335:       }
336: 
337:       if( count($paginas) > 1 )
338:       {
339:          return $paginas;
