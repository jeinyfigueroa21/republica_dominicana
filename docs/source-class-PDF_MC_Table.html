 333:     function NbLines($w,$txt)
 334:     {
 335:         //Computes the number of lines a MultiCell of width w will take
 336:         $cw=&$this->CurrentFont['cw'];
 337:         if($w==0)
 338:         $w=$this->w-$this->rMargin-$this->x;
 339:         $wmax=($w-2*$this->cMargin)*1000/$this->FontSize;
 340:         $s=str_replace("\r",'',$txt);
 341:         $nb=strlen($s);
 342:         if($nb>0 and $s[$nb-1]=="\n")
 343:         $nb--;
 344:         $sep=-1;
 345:         $i=0;
 346:         $j=0;
 347:         $l=0;
 348:         $nl=1;
 349:         while($i<$nb)
 350:         {
 351:             $c=$s[$i];
 352:             if($c=="\n")
 353:             {
 354:                 $i++;
 355:                 $sep=-1;
 356:                 $j=$i;
 357:                 $l=0;
 358:                 $nl++;
 359:                 continue;
 360:             }
 361:             if($c==' ')
 362:                 $sep=$i;
 363:             $l+=$cw[$c];
 364:             if($l>$wmax)
 365:             {
 366:                 if($sep==-1)
 367:                 {
 368:                     if($i==$j)
 369:                     $i++;
 370:                 }
 371:                 else
 372:                 $i=$sep+1;
 373:                 $sep=-1;
 374:                 $j=$i;
 375:                 $l=0;
 376:                 $nl++;
 377:             }
 378:             else
 379:                 $i++;
 380:         }
 381:         return $nl;
 382:     }

 384:     function RoundedRect($x, $y, $w, $h,$r, $style = '')
 385:     {
 386:         $k = $this->k;
 387:         $hp = $this->h;
 388:         if($style=='F')
 389:             $op='f';
 390:         elseif($style=='FD' or $style=='DF')
 391:             $op='B';
 392:         else
 393:             $op='S';
 394:         $MyArc = 4/3 * (sqrt(2) - 1);
 395:         $this->_out(sprintf('%.2f %.2f m',($x+$r)*$k,($hp-$y)*$k ));
 396:         $xc = $x+$w-$r ;
 397:         $yc = $y+$r;
 398:         $this->_out(sprintf('%.2f %.2f l', $xc*$k,($hp-$y)*$k ));
 399: 
 400:         $this->_Arc($xc + $r*$MyArc, $yc - $r, $xc + $r, $yc - $r*$MyArc, $xc + $r, $yc);
 401:         $xc = $x+$w-$r ;
 402:         $yc = $y+$h-$r;
 403:         $this->_out(sprintf('%.2f %.2f l',($x+$w)*$k,($hp-$yc)*$k));
 404:         $this->_Arc($xc + $r, $yc + $r*$MyArc, $xc + $r*$MyArc, $yc + $r, $xc, $yc + $r);
 405:         $xc = $x+$r ;
 406:         $yc = $y+$h-$r;
 407:         $this->_out(sprintf('%.2f %.2f l',$xc*$k,($hp-($y+$h))*$k));
 408:         $this->_Arc($xc - $r*$MyArc, $yc + $r, $xc - $r, $yc + $r*$MyArc, $xc - $r, $yc);
 409:         $xc = $x+$r ;
 410:         $yc = $y+$r;
 411:         $this->_out(sprintf('%.2f %.2f l',($x)*$k,($hp-$yc)*$k ));
 412:         $this->_Arc($xc - $r, $yc - $r*$MyArc, $xc - $r*$MyArc, $yc - $r, $xc, $yc - $r);
