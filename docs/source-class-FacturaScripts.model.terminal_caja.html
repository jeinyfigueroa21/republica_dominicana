226:    public function center_text($word = '', $ancho = FALSE)
227:    {
228:       if(!$ancho)
229:       {
230:          $ancho = $this->anchopapel;
231:       }
232: 
233:       if( strlen($word) == $ancho )
234:       {
235:          return $word;
236:       }
237:       else if( strlen($word) < $ancho )
238:       {
239:          return $this->center_text2($word, $ancho);
240:       }
241:       else
242:       {
243:          $result = '';
244:          $nword = '';
245:          foreach( explode(' ', $word) as $aux )
246:          {
247:             if($nword == '')
248:             {
249:                $nword = $aux;
250:             }
251:             else if( strlen($nword) + strlen($aux) + 1 <= $ancho )
252:             {
253:                $nword = $nword.' '.$aux;
254:             }
255:             else
256:             {
257:                if($result != '')
258:                {
259:                   $result .= "\n";
260:                }
261: 
262:                $result .= $this->center_text2($nword, $ancho);
263:                $nword = $aux;
264:             }
265:          }
266:          if($nword != '')
267:          {
268:             if($result != '')
269:             {
270:                $result .= "\n";
271:             }
272: 
273:             $result .= $this->center_text2($nword, $ancho);
274:          }
275: 
276:          return $result;
277:       }
278:    }

280:    private function center_text2($word = '', $ancho = 40)
281:    {
282:       $symbol = " ";
283:       $middle = round($ancho / 2);
284:       $length_word = strlen($word);
285:       $middle_word = round($length_word / 2);
286:       $last_position = $middle + $middle_word;
287:       $number_of_spaces = $middle - $middle_word;
288:       $result = sprintf("%'{$symbol}{$last_position}s", $word);
289:       for($i = 0; $i < $number_of_spaces; $i++)
290:       {
291:          $result .= "$symbol";
292:       }
293:       return $result;
294:    }
